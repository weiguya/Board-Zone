document.getElementById('createRoomForm').addEventListener('submit', (e) => { e.preventDefault(); const roomName = document.getElementById('roomName').value.trim(); const game = document.getElementById('game').value; const maxPlayers = document.getElementById('maxPlayers').value; if (roomName && game && maxPlayers) { // เก็บข้อมูลใน Local Storage localStorage.setItem('roomName', roomName); // ส่งข้อมูลไปที่เซิร์ฟเวอร์ socket.emit('create room', { roomName, game, maxPlayers }); // พาไปหน้า Waiting Room window.location.href = '/waiting-room.html'; } else { alert('Please fill in all fields!'); } });
